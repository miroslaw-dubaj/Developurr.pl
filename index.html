Wczytuje liczbę całkowitą – rok, a następnie sprawdza czy jest to rok przestępny i wyświetla stosowną informację. (Rok przestępny: dzieli się przez 4 i nie dzieli się przez 100, lub dzieli się przez 400).

Podaj rok do sprawdzenia:
Rozwiązanie

Tu pojawi się rozwiązanie

function solZad4() {
  let year = document.getElementById("l_year");
  let przest = "";
  let war1 = ((year % 4 === 0) && (year % 100 !== 0));
  let war2 = ((year % 400) === 0);
  if ( war1 || war2 ) {
    przest = "Rok przestępny";
  } else  {
    przest = "Rok nieprzestępny";
  }
document.getElementById("sol-zad-4").innerHTML = przest;
}

Zadanie 6 |

Do pewnego wydarzenia pozostało N milisekund, przekonwertować to do H:M:S (taki timer) 3723000 -> 1h 2m 3s

Podaj ile milisekund pozostało:
Rozwiązanie

Tu pojawi się rozwiązanie

function solZad6() {

  let msFromForm = document.getElementById("l_5");
  let ms = Number(msFromForm.value);
  let H;
  let M;
  let S;

  H = Math.floor(ms / 3600000);
  M = (ms - (H * 3600000));
  M = Math.floor(M / 60000);
  S = (ms - ((H *3600000) + (M * 60000)));
  S = Math.floor(S / 1000);

document.getElementById("sol-zad-4").innerHTML = ("Zostało" + H + "godzin," + M + "minut i " + S + "sekund." );
}
